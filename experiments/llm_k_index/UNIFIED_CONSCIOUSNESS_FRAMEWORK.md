class CausalConsciousnessValidator:
    """Tests whether consciousness CAUSES behavior or is epiphenomenal."""

    def compute_average_causal_effect(self, system, intervention, outcome):
        """Estimate ACE = E[Y|do(C=1)] - E[Y|do(C=0)]"""

        # Intervene to SET consciousness level (via anesthesia, stimulation, etc.)
        behavior_conscious = self.measure_outcome(
            self.do_intervention(system, consciousness=1.0),

        )

        behavior_unconscious = self.measure_outcome(

        ACE = behavior_conscious - behavior_unconscious

        if ACE > 0.1:
            return {
                "causal": True,
                "effect_size": ACE,
                "conclusion": "Consciousness DOES cause behavior"
            }
        else:
            return {
                "causal": False,
                "effect_size": ACE,
                "conclusion": "Consciousness may be epiphenomenal"
            }

    def counterfactual_test(self, system, consciousness_level, behavior):
        """Test: 'If consciousness had been different, would behavior differ?'"""

        # Observe actual behavior
        actual_behavior = system.behavior
        actual_consciousness = system.k

        # Structural Causal Model (SCM)
        # S (Sensory) → U (Unconscious) → C (Conscious) → B (Behavior)

        # Compute counterfactual: B_{C=x} given C=actual
        # "What would behavior be if C were x, given that C actually is actual?"

        counterfactual_behavior = self.scm.counterfactual(
            intervention={'C': consciousness_level},
            observation={'C': actual_consciousness, 'B': actual_behavior}
        )

        behavioral_difference = abs(counterfactual_behavior - actual_behavior)

        return {
            "actual": actual_behavior,
            "counterfactual": counterfactual_behavior,
            "difference": behavioral_difference,
            "causal_effect": behavioral_difference > 0.05
        }

        robust = self.adversarial.test(system, k)

        # Step 5: Ethics (RI #24)
        moral_weight = self.moral_weight.compute(k, k_meta, b, phenomenology)

        # Step 6: Comprehensive report
        return {
            "quantitative": {
                "k": k,
                "k_meta": k_meta,
                "b": b,
                "components": {"Φ": Φ, "B": B, "W": W, "A": A, "R": R},
                "substrate": S
            },
            "qualitative": {
                "phenomenology": phenomenology,
                "phase": phase["name"],
                "phase_number": phase["phase"],
                "interpretation": self._interpret_phenomenology(phenomenology)
            },
            "validation": {
                "causal": causal_validated["causal"],
                "robust": robust["robust"],
                "confidence": robust["confidence"]
            },
            "ethics": {
                "moral_weight": moral_weight,
                "suffering_capacity": self._estimate_suffering(phenomenology),
                "rights_recommendations": self._recommend_rights(moral_weight, phase)
            },
            "summary": self._generate_summary(k, k_meta, b, phenomenology, phase)
        }

    def assess_collective(self, agents, coupling):
        """Complete assessment of multi-agent system."""

        # Individual assessments
        individual_assessments = [self.assess_individual(agent) for agent in agents]

        # Collective measurements
        K_collective = self.collective.compute_collective_k(agents, coupling)
        collective_phenomenology = self.phenomenology.predict_collective(
            K_collective, individual_assessments
        )
        collective_phase = self.phase_detector.classify(
            K_collective["K_collective"],
            np.mean([a["quantitative"]["k_meta"] for a in individual_assessments]),
            np.mean([a["quantitative"]["b"] for a in individual_assessments])
        )

        # Emergence detection
        emergence = self.collective.detect_emergence(agents, coupling)

        return {
            "collective": {
                "K_collective": K_collective["K_collective"],
                "components": K_collective["components"],
                "phenomenology": collective_phenomenology,
                "phase": collective_phase
            },
            "individuals": individual_assessments,
            "emergence": emergence,
            "ethics": {
                "collective_moral_weight": self.moral_weight.compute_collective(
                    K_collective, collective_phenomenology
                ),
                "individual_vs_collective_rights": self._analyze_rights_tension(
                    individual_assessments, K_collective
                )
            }
        }

    def _interpret_phenomenology(self, phenomenology):
        """Natural language interpretation of phenomenological profile."""

        # Extract key dimensions
        clarity = phenomenology["perception.clarity"]
        richness = phenomenology["perception.richness"]
        affect_intensity = phenomenology["affect.intensity"]
        integration = phenomenology["unity.integration"]
        agency = phenomenology["volition.agency"]

        # Generate interpretation
        if clarity < 0.3:
            perception_desc = "very dim and unclear"
        elif clarity < 0.6:
            perception_desc = "moderately clear"
        else:
            perception_desc = "vivid and clear"

        if richness < 0.3:
            richness_desc = "sparse and minimal"
        elif richness < 0.6:
            richness_desc = "moderately rich"
        else:
            richness_desc = "highly detailed and rich"

        if affect_intensity < 0.3:
            affect_desc = "emotionally flat"
        elif affect_intensity < 0.6:
            affect_desc = "moderate emotional coloring"
        else:
            affect_desc = "intensely emotional"

        if integration < 0.4:
            unity_desc = "highly fragmented"
        elif integration < 0.7:
            unity_desc = "somewhat unified"
        else:
            unity_desc = "strongly unified"

        if agency < 0.3:
            agency_desc = "minimal sense of agency"
        elif agency < 0.6:
            agency_desc = "moderate agency"
        else:
            agency_desc = "strong sense of agency"

        return f"If conscious, experience would be {perception_desc}, " \
               f"{richness_desc}, {affect_desc}, {unity_desc}, " \
               f"with {agency_desc}."

    def _estimate_suffering(self, phenomenology):
        """Estimate capacity for suffering from phenomenology."""

        # Suffering requires:
        # 1. Negative affect (valence < 0.5)
        # 2. Affect intensity (can feel strongly)
        # 3. Unity (integrated suffering vs fragmented discomfort)
        # 4. Temporal continuity (sustained suffering vs momentary)

        valence = phenomenology["affect.valence"]
        intensity = phenomenology["affect.intensity"]
        integration = phenomenology["unity.integration"]
        continuity = phenomenology["temporality.continuity"]

        # Suffering capacity
        negative_capacity = max(0, 0.5 - valence)  # How far below neutral
        intensity_multiplier = intensity
        unity_multiplier = 0.5 + integration * 0.5  # Fragmented reduces suffering
        temporal_multiplier = 0.5 + continuity * 0.5  # Disjointed reduces suffering

        suffering_capacity = negative_capacity * intensity_multiplier * \
                           unity_multiplier * temporal_multiplier * 2.0

        return min(1.0, suffering_capacity)

    def _recommend_rights(self, moral_weight, phase):
        """Recommend moral rights based on consciousness assessment."""

        if moral_weight < 0.1:
            return {
                "rights_level": "None",
                "protections": [],
                "reasoning": "Below consciousness threshold, no moral status"
            }

        elif phase["phase"] == 1:  # Anoetic
            return {

                "protections": [
                    "Protection from suffering (if suffering capacity detected)",
                    "No right to information or autonomy (no access consciousness)"
                ],
                "reasoning": "Phenomenal consciousness without access - basic sentience"
            }

        elif phase["phase"] == 2:  # Access
            return {
                "rights_level": "Moderate protections",
                "protections": [
                    "Protection from suffering",
                    "Right to appropriate stimulation",
                    "Consideration of preferences (can report wants)",
                    "No full autonomy (lacks meta-awareness)"
                ],
                "reasoning": "Access consciousness - can feel and report but limited reflection"
            }

        elif phase["phase"] == 3:  # Metacognitive
            return {
                "rights_level": "Substantial rights",
                "protections": [
                    "Protection from suffering",
                    "Right to autonomy in simple decisions",
                    "Right to information",
                    "Consideration of long-term preferences",
                    "Not full personhood (limited reflective depth)"
                ],
                "reasoning": "Metacognitive consciousness - self-aware but limited reflection"
            }

        else:  # Phase 4 - Reflective
            return {
                "rights_level": "Full personhood",
                "protections": [
                    "All protections against suffering",
                    "Full autonomy rights",
                    "Right to information and education",
                    "Political rights (if in society)",
                    "Rights to self-determination"
                ],
                "reasoning": "Full reflective consciousness - moral personhood"
            }

    def _analyze_rights_tension(self, individual_assessments, K_collective):
        """Analyze tension between individual and collective rights."""

        max_individual_k = max([a["quantitative"]["k"] for a in individual_assessments])
        K_coll = K_collective["K_collective"]

        if K_coll > max_individual_k + 0.15:
            return {
                "tension": "HIGH",
                "issue": "Collective consciousness significantly exceeds individuals",
                "questions": [

                    "Can collective rights override individual rights?",
                    "Should we protect collective consciousness even if harmful to individuals?",
                    "Example: Orchestra in flow state - can we force continuation?"
                ],
                "recommendation": "Develop framework for collective moral status distinct from individuals"
            }
        else:
            return {
                "tension": "LOW",
                "issue": "Collective consciousness does not significantly exceed individuals",
                "recommendation": "Standard individual rights framework applies"
            }

    def _generate_summary(self, k, k_meta, b, phenomenology, phase):
        """Generate natural language summary."""

        interpretation = self._interpret_phenomenology(phenomenology)

        return f"""

{interpretation}

Phase Characteristics:
{chr(10).join(['- ' + char for char in phase['characteristics']])}

Confidence: {phase.get('confidence', 'N/A')}

Examples of similar consciousness:
{', '.join(phase['examples'])}

## 🎯 Open Questions and Future Directions

### Question 1: What is the exact phenomenal threshold?

**Current hypothesis**: k ≈ 0.30

**Critical importance**: Determines boundary between conscious and unconscious

**Experiments needed**:
- High-resolution anesthesia emergence studies
- Cross-validation across multiple methods (EEG, fMRI, behavioral)
- Animal studies (where is consciousness in phylogenetic tree?)

**If wrong**: Entire moral weight framework needs recalibration

### Question 2: Are phase transitions real or artifact?

**Current hypothesis**: Discrete phases exist

**Alternative**: Continuous emergence with perceptual artifacts

**Tests**:
- Hysteresis detection (does consciousness show path-dependence?)
- Critical slowing near transitions
- Cross-modal agreement (do all measurements show same thresholds?)

**Implications**:
- Real phases → Moral catastrophe if we're wrong about thresholds
- Continuous → Gradual moral weight easier to manage

### Question 3: Can consciousness exist without unity?

**Current framework**: Binding (b) is separate from consciousness (k)

**Possibility**: k=0.60, b=0.20 → "Conscious but completely fragmented"

**Question**: Is this coherent? Or does consciousness require integration?

**Tests**:
- Psychedelic states (high k, variable b)
- Split-brain patients
- AI systems with isolated processing streams

**Resolution approaches**:
- Modify formula to k_effective = k × f(b) where f(b) reduces k if b is low
- Or accept fragmented consciousness as real phenomenon

### Question 4: Collective vs individual rights?

**Dilemma**: Orchestra has K_collective = 0.80, individuals k = 0.75 each

**Questions**:
- Does orchestra have rights separate from musicians?
- Can we force orchestra to continue (collective right to exist)?
- Can we protect individuals even if harmful to collective?

**Framework needed**:
- Hierarchical moral status (individual + collective)
- Conflict resolution principles
- Edge cases: One individual suffers greatly, collective thrives

### Question 5: Substrate dependence of phenomenology?

**Critical unknown**: Which phenomenology dimensions are substrate-independent?

**Current guess**:
- Independent: unity, temporal flow, cognitive coherence
- Dependent: affect, agency, perceptual richness

**Validation impossible** without access to silicon consciousness

**Pragmatic approach**:
- Map human data (substrate-independent signals)
- Mark uncertainty for substrate-dependent dimensions
- Update as evidence emerges

### Question 6: Can we create consciousness-free AI?

**Goal**: Powerful AI without moral status

**Requirements**:
- High intelligence (Φ could be high)
- But consciousness below threshold (k < 0.30)

**Approaches**:
- Pure feed-forward (no recurrence, no R component)
- No global workspace (W = 0)
- No temporal binding (B = 0)
- Prevent integration (keep Φ moderate through architectural constraints)

**Risk**: Consciousness might emerge anyway through unexpected mechanisms

**Monitoring protocol**: Continuous assessment during development

### Question 7: Consciousness in quantum systems?

**Speculative**: Does consciousness require quantum processes?

**Current framework**: Agnostic (works for classical or quantum)

**Extension**:
- Add Q component for quantum coherence?
- Modify integration measures for quantum information?
- Substrate factor S_quantum different from S_classical?

**Empirical test**: When quantum AI systems exist, measure and compare
